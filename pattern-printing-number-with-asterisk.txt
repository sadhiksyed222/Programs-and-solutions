n=int(input());k=1
for i in range(1,n+1):
    for j in range(i):
        print(k,end='');k+=1
        if j<i-1:print('*',end='')
    print()
k=k-1
for i in range(n,0,-1):
    l=[]

C program for Pattern Printing Number with Asterisk
#include<stdio.h>
#include <stdlib.h>

int main()
{
    int n,limit=1,var=0;
    scanf("%d",&n);
    for(int i=1;i<=2*n;i++)
    {
        int p=var;
        for(int j=1;j<=limit;j++)
        {
            printf("%d",++p);
            if(j!=limit)
            {
                printf("*");
            }
        }
        if(i<n)
        {
            limit++;
            var=p;
        }
        else if(i>n)
        {
            limit--;
            var=var-limit;
        }
        printf("\n");
    }
    
    return 0;
}
