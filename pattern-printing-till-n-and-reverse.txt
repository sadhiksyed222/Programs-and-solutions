Input Format:
The first line contains N.
Output Format:
2N lines as shown in the Example Input/Output.
Boundary Conditions:
2 <=  N <= 100

n=int(input());p=[]
for i in range(1,n+1):
  l=[]
  for j in range(i):l.append(i)
  for i in l:print(i,end='')
  print();p.append(l)
for i in p[::-1]:

#include<stdio.h>
#include <stdlib.h>

int main()
{
int n,a=1,i,j;
scanf("%d",&n);
for(i=0;i<n;i++)
{
    for(j=0;j<=i;j++)
        printf("%d",a);
    a++;
    printf("\n");
}
for(i=0;i<n;i++)
{
    a--;
    for(j=0;j<n-i;j++)
        printf("%d",a);
    printf("\n");
}

}
