Given an integer N as the input, the program must print 2N lines pattern output as described in the Example Input/Output given below.
Input Format:
The first line contains N.
Output Format:
2N lines of the pattern.
Boundary Conditions:
2 <= N <= 100

4

#include <stdio.h>

int main() {
    int n,i,j;
    scanf("%d",&n);
    for(i=1;i<=2*n;i++){
        if(i<=n)
        for(j=0;j<i;j++)
        printf("%d",i);
        else
        for(j=0;j<2*n-i+1;j++)
        printf("%d",2*n-i+1);
        printf("\n");
    }
}
#include <iostream>

using namespace std;

{
int n,i,j;
