unsigned long long int getIntegerWithDigitsCount(int N)
{
  unsigned long long int m=N,k,l=0,a=0,count;
  while(m>0)
  {
      count=1;
      k=m/10;
      while(k>0)
      {
          if(m%10==k%10)
          count++;
          else
          break;
          k/=10;
      }
      l+=(m%10)*pow(10,a);
      a++;
      l+=(count)*pow(10,a);
      a++;
      m=k;
  }
  return l;
}
